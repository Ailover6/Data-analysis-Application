import tkinter as tk
from tkinter import filedialog, messagebox
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

def load_data():
    global data
    file_path = filedialog.askopenfilename(title="Select Data File", filetypes=[("CSV files", "*.csv"), ("Excel files", "*.xlsx")])
    if not file_path:
        return
    
    try:
        if file_path.endswith('.csv'):
            data = pd.read_csv(file_path)
        else:
            data = pd.read_excel(file_path)
        messagebox.showinfo("Success", "Data loaded successfully!")
    except Exception as e:
        messagebox.showerror("Error", f"Could not load file: {e}")

def show_summary():
    if data is None:
        messagebox.showwarning("Warning", "Please load a dataset first.")
        return
    summary = data.describe()
    messagebox.showinfo("Data Summary", summary.to_string())

def check_missing_values():
    if data is None:
        messagebox.showwarning("Warning", "Please load a dataset first.")
        return
    missing = data.isnull().sum()
    messagebox.showinfo("Missing Values", missing.to_string())

def plot_correlation():
    if data is None:
        messagebox.showwarning("Warning", "Please load a dataset first.")
        return
    
    plt.figure(figsize=(8, 6))
    sns.heatmap(data.corr(), annot=True, cmap="coolwarm", fmt=".2f")
    plt.title("Correlation Heatmap")
    plt.show()

data = None

root = tk.Tk()
root.title("Data Analysis Tool")
root.geometry("500x300")

frame = tk.Frame(root)
frame.pack(pady=20)

load_button = tk.Button(frame, text="Load Data", command=load_data)
summary_button = tk.Button(frame, text="Show Summary", command=show_summary)
missing_button = tk.Button(frame, text="Check Missing Values", command=check_missing_values)
correlation_button = tk.Button(frame, text="Plot Correlation", command=plot_correlation)

load_button.grid(row=0, column=0, padx=10)
summary_button.grid(row=0, column=1, padx=10)
missing_button.grid(row=1, column=0, padx=10)
correlation_button.grid(row=1, column=1, padx=10)

root.mainloop()
